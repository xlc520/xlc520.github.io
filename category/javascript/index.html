JavaScript数组去重方法合集

01. 双重 for 循环


var arr = [2,2,3,8,2,2,5,5,12,33,43,5,4,15,9];
function deduplication(arr){
   for (var i = 0; i < arr.length; i++) {
       for (var j = 0; j < arr.length; j++) {
           if (arr[i] == arr[j] && i != j) { 
              arr.splice(j, 1);
            }
       }
    }
    return arr;
}
var arr2  = deduplication(arr);
console.log(arr2);
